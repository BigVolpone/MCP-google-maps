# Étape 1 : Construire l'application
FROM node:18-alpine AS builder
WORKDIR /app

# Copier les fichiers nécessaires
COPY ./package.json ./package-lock.json ./
COPY ./tsconfig.json ./
COPY ./*.ts ./  # Copier tous les fichiers TypeScript dans la racine

# Installer les dépendances et construire le projet
RUN npm install
RUN npx tsc  # Compiler les fichiers TypeScript

# Étape 2 : Construire l'image de production
FROM node:18-alpine
WORKDIR /app

# Copier uniquement les fichiers nécessaires pour l'exécution
COPY --from=builder /app/dist ./dist
COPY ./package.json ./package-lock.json ./

# Installer uniquement les dépendances de production
RUN npm ci --omit=dev

# Démarrer le serveur MCP
CMD ["node", "dist/index.js"]
